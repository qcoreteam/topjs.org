(static) memoize(func, scope, hashFunc) → {function}

Source:
  • lang/Function.js, line 520

返回一个代理函数，这个代理函数会针对相应的参数对函数的执行结果进行缓存，当以同样的参数调用函数 的时候函数直接返回缓存的结果


function factorial (value) {
   let ret = value;
   while (--value > 1) {
      ret *= value;
   }
   return ret;
}

//创建缓存函数的包装器
factorial = TopJs.Function.memoize(factorial);
let val1 = factorial(30);
let val2 = factorial(30);
//val2直接使用缓存的结果

//支持多个参数调用
function permutation(n, k)
{
   return factorial(n) / factorial(n - k);
}

permutation = TopJs.Function.permutation(permutation, null, function(n, k){
   return `${n}-${k}`
});

特别注意
缓存是存在内存里面，千万不要使用不一样的参数进行长时间调用，否则内存最终会溢出
